parent_has_genetic_trait = {
	OR = {
		AND = {
			exists = mother
			mother = { has_trait_with_flag = genetic_trait }
		}
		AND = {
			exists = father
			father = { has_trait_with_flag = genetic_trait }
		}
		AND = {
			exists = real_father
			NOT = { real_father = father }
			real_father = { has_trait_with_flag = genetic_trait }
		}
	}
}
